Python Integration
import requests

url = "http://127.0.0.1:5021/api/predict"
headers = {
    "Content-Type": "application/json",
    "X-API-Key": "sdn_monitor_2024"
}

flow = {
    "switch": 1, "pktcount": 4777,
    "bytecount": 5092282, "dur": 10,
    "dur_nsec": 711000000, "flows": 3,
    "pktrate": 0, "Pairflow": 0,
    "port_no": 2, "tx_bytes": 3753,
    "rx_bytes": 1332, "tx_kbps": 0,
    "rx_kbps": 0
}

r = requests.post(url, json=flow,
                   headers=headers)
print(r.json())
# {"result": "DDoS", ...}
